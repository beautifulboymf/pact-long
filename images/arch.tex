% Figure 2 — PACT framework pipeline with inline theory callouts.
% Render via \input{images/arch.tex} inside a figure* environment.
% Requires (in main.tex preamble):
%   \usepackage{tikz}
%   \usetikzlibrary{arrows.meta, fit, positioning, calc, backgrounds}
%   \definecolor{pactaccent}{HTML}{2A8A8A}   % teal  — bias-channel
%   \definecolor{varaccent}{HTML}{8A4FAD}    % purple — variance-channel
\begin{tikzpicture}[
    every node/.style={font=\small},
    %% generic pipeline blocks
    blk/.style={
        draw, rounded corners=2pt, align=center,
        minimum width=1.85cm, minimum height=0.95cm,
        inner sep=3pt, line width=0.55pt, draw=black!55, fill=black!2,
    },
    biasblk/.style={
        blk, draw=pactaccent!85, fill=pactaccent!12, line width=0.85pt,
    },
    varblk/.style={
        blk, draw=varaccent!85, fill=varaccent!10, line width=0.85pt,
    },
    %% arrows
    pipe/.style={-Stealth, line width=0.55pt, draw=black!70},
    biaspipe/.style={-Stealth, line width=0.75pt, draw=pactaccent},
    varpipe/.style={-Stealth, line width=0.75pt, draw=varaccent},
    detach/.style={-Stealth, dashed, line width=0.7pt, draw=red!65!black},
    %% callout bubbles
    callout/.style={
        rounded corners=3pt, align=center, inner sep=5pt,
        line width=0.7pt, text width=3.2cm,
    },
    biascall/.style={callout, draw=pactaccent!85, fill=pactaccent!6,
        font=\scriptsize},
    varcall/.style={callout, draw=varaccent!85, fill=varaccent!6,
        font=\scriptsize},
    %% leader lines from callout to plug-in
    leader/.style={line width=0.6pt, densely dashed, draw=black!35},
    %% latent node used inside callouts
    tinyZ/.style={draw, circle, fill=black!4, minimum size=0.38cm,
        inner sep=0pt, line width=0.5pt, font=\scriptsize},
    tinybox/.style={draw, rounded corners=1pt, inner sep=1.2pt,
        font=\scriptsize, line width=0.4pt, minimum height=0.3cm},
    %% misc
    plugin/.style={
        font=\scriptsize\bfseries, inner sep=2pt, outer sep=0pt,
        rounded corners=2pt, fill=white,
    },
    sublabel/.style={font=\scriptsize\itshape, text=black!55, align=center},
    rhead/.style={font=\small\bfseries, text=black!78},
]

% ===================================================================
%   BOTTOM ROW — linear training pipeline
% ===================================================================
\node[blk] (input) at (0, 0) {$X,\,\mathcal{G}$};

\node[biasblk, right=0.55cm of input] (gpe)
    {\textbf{GPE}\\[1pt]{\scriptsize$X \oplus p(Z^*)$}\\[-1pt]{\scriptsize node2vec $+$ cross-attn}};

\node[blk, right=0.55cm of gpe, minimum width=2.2cm] (M)
    {base model $M$};

\node[blk, right=0.55cm of M] (mu) {$\hat\mu_0,\,\hat\mu_1$};

\node[blk, right=0.55cm of mu, minimum width=2.2cm] (pseudo)
    {$D = Y - \hat\mu_{1-t}$};

\node[varblk, right=0.55cm of pseudo, minimum width=2.8cm] (loss)
    {\textbf{Variance-Weighted Loss}\\[1pt]{\scriptsize$\sum_i w_i(D_i - \hat\tau)^2$}\\[-1pt]
     {\scriptsize$w_i = 1/\max\{\hat\sigma^2_t,\delta\}$}};

% Pipeline edges.
\draw[biaspipe] (input) -- (gpe);
\draw[pipe]     (gpe)   -- (M);
\draw[pipe]     (M)     -- (mu);
\draw[pipe]     (mu)    -- (pseudo);
\draw[pipe]     (pseudo) -- (loss);

% Backbone list and variance-head callout below the pipeline.
\node[sublabel, below=0.08cm of M, text width=2.6cm]
    {BNN, TARNet, NetDeconf,\\GIAL, GNUM, GDC, X-learner};

\node[varblk, below=0.6cm of loss, minimum width=2.2cm, minimum height=0.6cm]
    (sh) {{\scriptsize$\log\hat\sigma^2_t$ head}};
\node[blk, dashed, draw=black!40, left=0.5cm of sh,
      minimum width=2.1cm, minimum height=0.6cm] (pilot)
    {{\scriptsize pilot $\hat\tau$ (detached)}};
\draw[detach]  (pilot.east) -- (sh.west);
\draw[varpipe] (sh.north) -- (sh.north |- loss.south);

% ===================================================================
%   TOP ROW — inline theory callouts, one per plug-in
% ===================================================================
\node[biascall, above=2.4cm of gpe, anchor=south] (biascall) {%
    \begin{tikzpicture}[baseline=(z.base)]
        \node[tinyZ] (z) {$Z^*$};
        \node[tinybox, right=0.18cm of z] (t) {$T$};
        \node[tinybox, below=0.02cm of t] (y) {$Y(t)$};
        \draw[biaspipe, line width=0.5pt] (z) -- (t);
        \draw[biaspipe, line width=0.5pt] (z) -- (y);
    \end{tikzpicture}\\[2pt]
    \textcolor{pactaccent}{\textbf{attacks bias$^2$}} — Prop.~\ref{prop:dualroot}(a)\\
    \textcolor{pactaccent}{\footnotesize\textbf{[ PACT plug-in ]}}%
};

\node[varcall, above=2.4cm of loss, anchor=south] (varcall) {%
    \begin{tikzpicture}[baseline=(z.base)]
        \node[tinyZ] (z) {$Z^*$};
        \node[tinybox, right=0.18cm of z] (s) {$\sigma^2(Y)$};
        \draw[varpipe, line width=0.5pt] (z) -- (s);
    \end{tikzpicture}\\[2pt]
    \textcolor{varaccent}{\textbf{attacks variance}} — Prop.~\ref{prop:dualroot}(b)\\
    \textcolor{varaccent}{\footnotesize\textbf{[ PACT plug-in ]}}%
};

% Leader lines from each callout to its plug-in.
\draw[leader, draw=pactaccent!70] (biascall.south) -- (gpe.north);
\draw[leader, draw=varaccent!80]  (varcall.south)  -- (loss.north);

% Strip label on the left — small, out of the flow.
\node[rhead, text=pactaccent, align=left, font=\scriptsize\bfseries]
    at ($(biascall.west)+(-1.6,0.0)$) {Theory \\ callouts:};
\node[rhead, align=left, font=\scriptsize\bfseries]
    at ($(input.west)+(-1.5,0.0)$) {Training \\ pipeline:};

\end{tikzpicture}
