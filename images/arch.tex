% Figure 2 — PACT architecture / training pipeline, v6.
% Render via \input{images/arch.tex} inside a figure* environment.
% Requires (in main.tex preamble):
%   \usepackage{tikz}
%   \usetikzlibrary{arrows.meta, fit, positioning, calc, backgrounds}
%   \definecolor{pactaccent}{HTML}{2A8A8A}   % teal  — bias channel (GPE)
%   \definecolor{varaccent}{HTML}{8A4FAD}    % purple — variance channel (VWL)
\begin{tikzpicture}[
    % ============== blocks ==============
    every node/.style={font=\small},
    %% small, neutral pipeline blocks (data / base model)
    blk/.style={
        draw, rounded corners=3pt, align=center,
        minimum width=1.9cm, minimum height=1.05cm,
        inner xsep=4pt, inner ysep=3pt,
        line width=0.6pt, draw=black!55, fill=black!3,
    },
    %% the base-model block is slightly wider and taller — it's the anchor
    modelblk/.style={
        blk, minimum width=2.3cm, minimum height=1.25cm,
        line width=0.8pt, fill=black!5,
    },
    %% GPE plug-in (teal) — prominent, designed to visually stand out
    gpeblk/.style={
        draw, rounded corners=3pt, align=center,
        minimum width=2.55cm, minimum height=1.55cm,
        inner xsep=5pt, inner ysep=4pt,
        line width=1.0pt,
        draw=pactaccent, fill=pactaccent!10,
    },
    %% VWL plug-in (purple) — same prominence as GPE
    vwlblk/.style={
        draw, rounded corners=3pt, align=center,
        minimum width=3.3cm, minimum height=1.55cm,
        inner xsep=5pt, inner ysep=4pt,
        line width=1.0pt,
        draw=varaccent, fill=varaccent!10,
    },
    %% sub-block for the log-sigma^2 head / pilot (smaller, lighter)
    subblk/.style={
        draw, rounded corners=2pt, align=center,
        minimum width=2.2cm, minimum height=0.75cm,
        inner xsep=3pt, inner ysep=2pt,
        line width=0.55pt,
        draw=varaccent!70, fill=varaccent!5,
        font=\scriptsize,
    },
    detachblk/.style={
        draw, rounded corners=2pt, align=center,
        minimum width=2.2cm, minimum height=0.75cm,
        inner xsep=3pt, inner ysep=2pt,
        line width=0.55pt, densely dashed,
        draw=black!45, fill=black!2,
        font=\scriptsize,
    },
    %% tags (plug-in badges above GPE / VWL)
    pactbadge/.style={
        rounded corners=2pt, inner xsep=4pt, inner ysep=1.5pt,
        line width=0.6pt, fill=white,
        font=\scriptsize\bfseries,
    },
    %% text labels
    sublbl/.style={
        font=\scriptsize\itshape, text=black!55, align=center,
    },
    phase/.style={
        font=\scriptsize\bfseries, text=black!60,
    },
    %% arrows
    pipe/.style={-Stealth, line width=0.65pt, draw=black!70},
    biaspipe/.style={-Stealth, line width=0.85pt, draw=pactaccent},
    varpipe/.style={-Stealth, line width=0.85pt, draw=varaccent},
    detach/.style={-Stealth, dashed, line width=0.75pt, draw=red!65!black},
    node distance=0.65cm,
]

% ============================== MAIN PIPELINE ==============================
% Input
\node[blk] (input) {$X,\;\mathcal{G}$};

% GPE plug-in — first contribution, input level
\node[gpeblk, right=of input] (gpe) {%
    \textbf{\textcolor{pactaccent}{GPE}}\\[2pt]
    {\small$X \oplus p(Z^*)$}\\[-1pt]
    {\scriptsize node2vec $+$ cross-attn}
};

% Base model
\node[modelblk, right=of gpe] (M) {%
    \textbf{base model}\\
    $M$
};
\node[sublbl, below=0.1cm of M, text width=2.5cm] (bblist)
    {BNN, TARNet, NetDeconf,\\GIAL, GNUM, GDC, X-learner};

% Outcome heads
\node[blk, right=of M] (mu) {$\hat\mu_0,\,\hat\mu_1$};

% Pseudo-outcome
\node[blk, right=of mu, minimum width=2.2cm] (pseudo)
    {$D\;=\;Y - \hat\mu_{1-t}$};

% VWL plug-in — second contribution, loss level
\node[vwlblk, right=of pseudo] (loss) {%
    \textbf{\textcolor{varaccent}{Variance-Weighted Loss}}\\[2pt]
    {\small$\sum_i w_i\,(D_i - \hat\tau)^2$}\\[-1pt]
    {\scriptsize$w_i = 1/\max\{\hat\sigma^2_t, \delta\}$}
};

% ---------- arrows along the main pipeline ----------
\draw[biaspipe] (input) -- (gpe);
\draw[pipe]     (gpe)   -- (M);
\draw[pipe]     (M)     -- (mu);
\draw[pipe]     (mu)    -- (pseudo);
\draw[varpipe]  (pseudo) -- (loss);

% ============================== SIGMA HEAD & PILOT ==============================
\node[subblk, below=0.8cm of loss]  (sh)    {$\log\hat\sigma^2_t$ head};
\node[detachblk, below=0.35cm of sh] (pilot) {pilot $\hat\tau$ (detached)};

% pilot -> head -> loss (weight flow)
\draw[detach]  (pilot.north) -- (sh.south);
\draw[varpipe] (sh.north) -- (loss.south);

% ============================== PLUG-IN BADGES ==============================
\node[pactbadge, draw=pactaccent, text=pactaccent,
      above=0.15cm of gpe] {PACT plug-in};
\node[pactbadge, draw=varaccent, text=varaccent,
      above=0.15cm of loss] {PACT plug-in};

% ============================== PHASE LABELS (optional subtle) ==============================
% Small left-margin labels indicating forward/backward roles.
\node[phase, anchor=west] at ($(input.west)+(-1.35,0.3)$) {forward pass};
\node[phase, anchor=west] at ($(input.west)+(-1.35,-0.3)$) {$\longrightarrow$};

% ============================== BACKGROUND BANDS ==============================
% Very subtle coloured bands behind GPE and VWL to emphasize where the
% plug-ins sit. Low opacity so they do not compete.
\begin{pgfonlayer}{background}
    \node[fill=pactaccent!5, rounded corners=6pt,
          fit=(gpe) (bblist), inner xsep=4pt, inner ysep=6pt,
          draw=pactaccent!15, line width=0.4pt] {};
    \node[fill=varaccent!5, rounded corners=6pt,
          fit=(loss) (pilot), inner xsep=4pt, inner ysep=6pt,
          draw=varaccent!18, line width=0.4pt] {};
\end{pgfonlayer}

\end{tikzpicture}
