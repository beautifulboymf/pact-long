% Figure 2 — PACT framework: theory strip (top) + training pipeline (bottom).
% Render via \input{images/arch.tex} inside a figure* environment.
% Requires (in main.tex preamble):
%   \usepackage{tikz}
%   \usetikzlibrary{arrows.meta, fit, positioning, calc, backgrounds}
%   \definecolor{pactaccent}{HTML}{2A8A8A}    % teal  - bias-channel
%   \definecolor{varaccent}{HTML}{8A4FAD}     % purple - variance-channel
\begin{tikzpicture}[
    every node/.style={font=\small},
    %% generic blocks
    blk/.style={
        draw, rounded corners=2pt, align=center,
        minimum width=1.9cm, minimum height=0.75cm,
        inner sep=3pt, line width=0.55pt, draw=black!55, fill=black!2,
    },
    biasblk/.style={
        blk, draw=pactaccent!85, fill=pactaccent!12, line width=0.85pt,
    },
    varblk/.style={
        blk, draw=varaccent!85,  fill=varaccent!10,  line width=0.85pt,
    },
    latent/.style={
        draw, circle, fill=black!4, align=center,
        minimum size=0.85cm, line width=0.8pt, inner sep=0pt,
    },
    %% arrows
    pipe/.style={-Stealth, line width=0.55pt, draw=black!70},
    biaspipe/.style={-Stealth, line width=0.75pt, draw=pactaccent},
    varpipe/.style={-Stealth, line width=0.75pt, draw=varaccent},
    detach/.style={-Stealth, dashed, line width=0.7pt, draw=red!65!black},
    reduce/.style={-Stealth, dash pattern=on 3pt off 2pt,
        line width=0.9pt},
    %% labels
    rhead/.style={font=\small\bfseries, text=black!78},
    plugin/.style={
        font=\scriptsize\bfseries, inner sep=2pt, outer sep=0pt,
        rounded corners=2pt,
    },
    sublabel/.style={font=\scriptsize\itshape, text=black!55, align=center},
    stripbg/.style={fill=black!3, draw=black!15, line width=0.35pt,
                    rounded corners=3pt, inner sep=6pt},
]

% ===================================================================
%   TOP STRIP — dual-root theory (Z* -> T, Y, sigma^2 -> PEHE^2)
% ===================================================================
\begin{scope}[local bounding box=theory]
    \node[rhead] (thdr) at (-7.2, 1.55) {\small Theory (Prop. 1):};

    % Latent Z* on the left of the strip.
    \node[latent] (Z) at (-5.3, 1.0) {$Z^*$};

    % Three children.
    \node[biasblk, minimum width=1.2cm] (T) at (-3.3, 1.55) {\small$T$};
    \node[biasblk, minimum width=1.2cm] (Yt) at (-3.3, 0.55) {\small$Y(t)$};
    \node[varblk,  minimum width=1.4cm] (Sg) at (-3.3, -0.45) {\small$\sigma^2(Y)$};

    \draw[biaspipe] (Z) -- (T);
    \draw[biaspipe] (Z) -- (Yt);
    \draw[varpipe]  (Z) -- (Sg);

    % Error-term aggregation.
    \node[biasblk, minimum width=1.6cm] (bias) at (-0.5, 1.05)
        {\small bias$^2$};
    \node[varblk,  minimum width=1.6cm] (var) at (-0.5, -0.45)
        {\small variance};

    \draw[biaspipe] (T.east)  -- (bias.west);
    \draw[biaspipe] (Yt.east) -- (bias.west);
    \draw[varpipe]  (Sg.east) -- (var.west);

    % PEHE total.
    \node[blk, minimum width=2.2cm, minimum height=1.1cm,
          align=center] (pehe) at (1.85, 0.3)
        {\small$\mathrm{PEHE}^2$};
    \draw[biaspipe] (bias.east) -- (pehe.west);
    \draw[varpipe]  (var.east)  -- (pehe.west);

    % "attack target" callouts that the bottom pipeline will point to.
    \coordinate (bias_anchor) at ($(bias.south) + (0, -0.04)$);
    \coordinate (var_anchor)  at ($(var.north)  + (0,  0.04)$);
\end{scope}
\begin{pgfonlayer}{background}
    \node[stripbg, fit=(theory), inner sep=10pt] (theorybg) {};
\end{pgfonlayer}

% ===================================================================
%   BOTTOM PIPELINE — training flow with two plug-ins
% ===================================================================
\begin{scope}[local bounding box=pipeline, yshift=-3.6cm]
    \node[rhead] (phdr) at (-7.2, 0.85) {\small PACT training pipeline:};

    % 1) Input
    \node[blk, minimum width=1.5cm, minimum height=0.95cm]
        (input) at (-5.5, 0) {$X,\,\mathcal{G}$};

    % 2) GPE (bias-channel plug-in)
    \node[biasblk, minimum width=1.9cm, minimum height=0.95cm]
        (gpe) at (-3.3, 0)
        {\textbf{GPE}\\{\scriptsize$X \oplus p(Z^*)$}\\{\scriptsize node2vec $+$ cross-attn}};
    \node[plugin, above=0.05cm of gpe, text=pactaccent,
          fill=pactaccent!14, draw=pactaccent]
        {PACT plug-in};
    \draw[biaspipe] (input) -- (gpe);

    % 3) Base model M
    \node[blk, minimum width=2.3cm, minimum height=0.95cm]
        (M) at (-0.7, 0) {base model $M$};
    \node[sublabel, below=0.05cm of M, text width=3.6cm]
        {BNN, TARNet, NetDeconf, GIAL, GNUM, GDC, X-learner};
    \draw[pipe] (gpe) -- (M);

    % 4) Outcome heads
    \node[blk, minimum width=1.6cm, minimum height=0.95cm]
        (mu) at (1.75, 0) {$\hat\mu_0,\hat\mu_1$};
    \draw[pipe] (M) -- (mu);

    % 5) Pseudo-outcome
    \node[blk, minimum width=2.0cm, minimum height=0.95cm]
        (pseudo) at (4.15, 0) {$D = Y - \hat\mu_{1-t}$};
    \draw[pipe] (mu) -- (pseudo);

    % 6) Variance-Weighted Loss (variance-channel plug-in)
    \node[varblk, minimum width=2.6cm, minimum height=0.95cm]
        (loss) at (7.0, 0)
        {\textbf{Variance-Weighted Loss}\\{\scriptsize$\sum_i w_i (D_i - \hat\tau)^2,\; w_i=\tfrac{1}{\max\{\hat\sigma^2_t, \delta\}}$}};
    \node[plugin, above=0.05cm of loss, text=varaccent,
          fill=varaccent!12, draw=varaccent]
        {PACT plug-in};
    \draw[pipe] (pseudo) -- (loss);

    % Variance head + detached pilot, drawn below the loss block
    \node[varblk, minimum width=1.8cm, minimum height=0.55cm]
        (sh) at (7.0, -1.25) {{\scriptsize$\log\hat\sigma^2_t$ head}};
    \node[blk, minimum width=2.2cm, minimum height=0.55cm,
          draw=black!40, dashed]
        (pilot) at (4.15, -1.25) {{\scriptsize pilot $\hat\tau$ (detached)}};
    \draw[varpipe]  (sh) -- (loss);
    \draw[detach]   (pilot) -- (sh);
\end{scope}

% ===================================================================
%   CROSS-LAYER "reduces" ARROWS  (bottom plug-ins -> top error terms)
% ===================================================================
% From GPE up to bias^2 box on the theory strip.
\draw[reduce, draw=pactaccent!85]
    (gpe.north) to[out=90, in=-90]
    node[font=\scriptsize\bfseries, text=pactaccent, sloped, above, pos=.55]
        {reduces bias$^2$ (a)}
    (bias_anchor);

% From Variance-Weighted Loss up to variance box.
\draw[reduce, draw=varaccent!85]
    (loss.north) to[out=90, in=-90]
    node[font=\scriptsize\bfseries, text=varaccent, sloped, above, pos=.55]
        {reduces variance (b)}
    (var_anchor);

\end{tikzpicture}
